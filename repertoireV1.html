df_na <- df %>% 
  mutate(across(
    .cols = where(is.character),
    .fns = ~ na_if(., "")
  ))

empty_cols <- df_na %>% 
  summarise(across(everything(), ~ sum(is.na(.)))) %>% 
  pivot_longer(everything(),
               names_to =  "variable",
               values_to = "nb_empty_cells") %>% 
  filter(nb_empty_cells == nrow(df_na))
